@equation.evolve
def equation(x1: np.ndarray, x2: np.ndarray, params: np.ndarray) -> np.ndarray:
    """ Mathematical function for a system with two inputs.

    Args:
        x1: A numpy array representing the first input variable.
        x2: A numpy array representing the second input variable.
        params: Array of numeric constants or parameters to be optimized

    Return:
        A numpy array representing the predicted output variable.
    """
    output = params[0] * x1 + params[1] * x2
